cd (Changes current directory)
git config --global user.name "your github username"
git config --global user.email "your github email ID" (Registers your device to git)
git init (initialises/reinitialises local repositories)
git remote add origin "URL" (Connects local repository to github repository)
git status (checks directory for changes not added, removed or committed)
git add/rem (adds or removes files pre-commit)
git commit -m "Message" (command for every commit)
git push -u origin main/master/"branchName" (Sends local commits upwards into the online github repository)
git checkout -b "branchName" (Creates a new branch and switches to it)
git checkout "branchName" (Switches to branchName)
git clone <repo url> (this downloads the online github repository locally into the current directory)
